import ifcopenshell
import ifcopenshell.util.element as Element
import pandas as pd

def extrair_quantitativos(modelo, tipo_elemento):
    """
    Extrai quantitativos de um tipo espec√≠fico de elemento.
    
    Args:
        modelo: Objeto IfcOpenShell do modelo
        tipo_elemento: String, ex: 'IfcWall', 'IfcSlab', 'IfcBeam'
    
    Returns:
        DataFrame com os quantitativos
    """
    elementos = modelo.by_type(tipo_elemento)
    dados = []
    
    for elem in elementos:
        # Obter propriedades b√°sicas
        info = {
            'ID': elem.GlobalId,
            'Nome': elem.Name if hasattr(elem, 'Name') else 'N/A',
            'Tipo': tipo_elemento
        }
        
        # Tentar obter quantidades (volume, √°rea, etc.)
        for definition in elem.IsDefinedBy:
            if definition.is_a('IfcRelDefinesByProperties'):
                property_set = definition.RelatingPropertyDefinition
                
                if property_set.is_a('IfcElementQuantity'):
                    for quantity in property_set.Quantities:
                        # Volume
                        if quantity.is_a('IfcQuantityVolume'):
                            info['Volume_m3'] = quantity.VolumeValue
                        # √Årea
                        elif quantity.is_a('IfcQuantityArea'):
                            info['Area_m2'] = quantity.AreaValue
                        # Comprimento
                        elif quantity.is_a('IfcQuantityLength'):
                            info['Comprimento_m'] = quantity.LengthValue
        
        dados.append(info)
    
    return pd.DataFrame(dados)

# Usar a fun√ß√£o
modelo = ifcopenshell.open('caminho/para/seu_modelo.ifc')

# Extrair quantitativos de lajes (concreto)

print("=== Quantitativos de Lajes ===")
print(f"Total de lajes: {len(lajes)}")

if 'Volume_m3' in lajes.columns:

Vamos calcular o carbono incorporado nos materiais do modelo:

import pandas as pd
import matplotlib.pyplot as plt

# Fatores de emiss√£o (kg CO2 por unidade)
# Fonte: ICE Database (Inventory of Carbon & Energy)
FATORES_CARBONO = {
    'concreto_c30': 350,      # kg CO2 por m¬≥
    'concreto_c40': 400,      # kg CO2 por m¬≥
    'aco': 2100,              # kg CO2 por tonelada
    'tijolo_ceramico': 0.24,  # kg CO2 por unidade
    'vidro': 850,             # kg CO2 por tonelada
    'aluminio': 8500,         # kg CO2 por tonelada
}

def calcular_carbono_concreto(volume_m3, tipo='concreto_c30'):
    """
    Calcula carbono incorporado em elementos de concreto.
    
    Args:
        volume_m3: Volume em metros c√∫bicos
        tipo: Tipo de concreto (concreto_c30 ou concreto_c40)
    
    Returns:
        Carbono em kg CO2
    """
    fator = FATORES_CARBONO.get(tipo, 350)
    carbono = volume_m3 * fator
    return carbono

def gerar_relatorio_carbono(modelo_path):
    """
    Gera relat√≥rio completo de carbono incorporado.
    """

        print("=== Relat√≥rio de Pegada de Carbono ===")
        print(f"Volume de concreto (lajes): {volume_total:.2f} m¬≥")
        print(f"Carbono incorporado: {carbono_lajes:,.0f} kg CO2")

        print(f"  - Equivalente a {carbono_lajes/120:,.0f} carros durante 1 m√™s")
        
        # Simular alternativa sustent√°vel

        print("Se substituirmos por concreto com 30% de esc√≥ria:")
        carbono_alternativo = volume_total * (350 * 0.7)  # 30% menos carbono

        print(f"Redu√ß√£o: {reducao:,.0f} kg CO2 ({reducao/carbono_lajes*100:.1f}%)")
        
        return {
            'volume': volume_total,
            'carbono_tradicional': carbono_lajes,
            'carbono_sustentavel': carbono_alternativo,
            'reducao': reducao
        }

import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error, r2_score

# Exemplo: Dados hist√≥ricos de projetos (simulados)
# Em um cen√°rio real, estes viriam de projetos anteriores
dados_historicos = pd.DataFrame({
    'area_construida_m2': [5000, 8000, 12000, 6500, 15000, 9000, 7500, 11000],
    'numero_pavimentos': [10, 15, 25, 12, 30, 18, 14, 22],
    'volume_concreto_m3': [2500, 4200, 7000, 3400, 9000, 5000, 3900, 6500],
    'tipo_padrao': [1, 1, 2, 1, 2, 2, 1, 2],  # 1=m√©dio, 2=alto
    'desperdicio_percentual': [8.5, 7.2, 5.1, 9.0, 4.8, 6.0, 7.8, 5.5]  # % de res√≠duo
})

# Features (caracter√≠sticas) e Target (alvo)
X = dados_historicos[['area_construida_m2', 'numero_pavimentos', 
                       'volume_concreto_m3', 'tipo_padrao']]
y = dados_historicos['desperdicio_percentual']

# Dividir em treino e teste
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42
)

# Treinar modelo
modelo_rf = RandomForestRegressor(n_estimators=100, random_state=42)
modelo_rf.fit(X_train, y_train)

# Avaliar
y_pred = modelo_rf.predict(X_test)
mae = mean_absolute_error(y_test, y_pred)
r2 = r2_score(y_test, y_pred)

print("=== Modelo de Previs√£o de Desperd√≠cio ===")
print(f"MAE (Mean Absolute Error): {mae:.2f}%")
print(f"R¬≤ Score: {r2:.3f}")

# Fazer previs√£o para um novo projeto
novo_projeto = pd.DataFrame({
    'area_construida_m2': [10000],
    'numero_pavimentos': [20],
    'volume_concreto_m3': [5500],
    'tipo_padrao': [2]  # Alto padr√£o
})

desperdicio_previsto = modelo_rf.predict(novo_projeto)[0]
print(f"\n=== Previs√£o para Novo Projeto ===")
print(f"√Årea: 10.000 m¬≤ | Pavimentos: 20 | Tipo: Alto Padr√£o")
print(f"Desperd√≠cio previsto: {desperdicio_previsto:.2f}%")

# Calcular impacto

custo_residuo = volume_residuo * 500  # R$ 500/m¬≥ (custo aproximado)

print(f"\nImpacto estimado:")
print(f"  - Volume de res√≠duo: {volume_residuo:.2f} m¬≥")
print(f"  - Custo do desperd√≠cio: R$ {custo_residuo:,.2f}")
print(f"  - Carbono desperdi√ßado: {volume_residuo * 350:.0f} kg CO2")

# Sugest√£o de melhoria
print(f"\nüí° Recomenda√ß√£o:")
print(f"Com uso de BIM+IA (clash detection), o desperd√≠cio pode ser reduzido para ~3%")

economia = custo_residuo - (volume_residuo_otimizado * 500)

print(f"Redu√ß√£o de carbono: {(volume_residuo - volume_residuo_otimizado) * 350:.0f} kg CO2")
